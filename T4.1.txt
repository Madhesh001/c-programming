#include <stdio.h>
int Composite(int n) 
{
  if (n <= 1) 
  {
  return 0;
  }
for (int i = 2; i <= n / 2; i++)
  {
   if (n % i == 0) 
    {
       return 1;
    }
  }
       return 0;

}
int main()
{
    int arr[] ={100, 100, 100, 100, 100, 100, 100, 100} ;
    int size = sizeof(arr) / sizeof(arr[0]);
    int compositeCount = 0;

for (int i = 0; i < size; i++) 
{
    if (Composite(arr[i])) 
    {
       compositeCount++;
    }

 }

    printf("Number of Composite Numbers = %d", compositeCount);

    return 0;

}
